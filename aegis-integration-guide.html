B.1Copy server files to ~/.aegis/mcp/▶

Copy the 5 server files from the aegis-mcp-layer spec into ~/.aegis/mcp/. Each file begins with a # FILE: ~/.aegis/mcp/X_server.py comment confirming its path.

cp ~/aegis-innovations/qdrant_server.py ~/.aegis/mcp/ cp ~/aegis-innovations/ollama_server.py ~/.aegis/mcp/ cp ~/aegis-innovations/fs_server.py ~/.aegis/mcp/ cp ~/aegis-innovations/system_server.py ~/.aegis/mcp/ cp ~/aegis-innovations/telegram_server.py ~/.aegis/mcp/ chmod +x ~/.aegis/mcp/*.py

⚠ Replace /home/ubuntu with your actual home directory. Check with: echo $HOME
cat > ~/.aegis/mcp_config.json << 'EOF' { "mcpServers": { "aegis-qdrant": { "command": "python3", "args": ["/home/ubuntu/.aegis/mcp/qdrant_server.py"], "transport": "stdio" }, "aegis-ollama": { "command": "python3", "args": ["/home/ubuntu/.aegis/mcp/ollama_server.py"], "transport": "stdio" }, "aegis-fs": { "command": "python3", "args": ["/home/ubuntu/.aegis/mcp/fs_server.py"], "transport": "stdio" }, "aegis-system": { "command": "python3", "args": ["/home/ubuntu/.aegis/mcp/system_server.py"], "transport": "stdio" }, "aegis-telegram": { "command": "python3", "args": ["/home/ubuntu/.aegis/mcp/telegram_server.py"], "transport": "stdio" } } } EOF

F.1Create nightly pipeline script▶
cat > ~/.aegis/scripts/nightly.sh << 'EOF' #!/bin/bash source ~/.aegis/.env LOG="$HOME/.aegis/logs/nightly_$(date +%Y%m%d).log" echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] Nightly pipeline start" | tee -a "$LOG" python3 ~/.aegis/context/context_engine.py consolidate 2>&1 | tee -a "$LOG" echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] Complete" | tee -a "$LOG" find ~/.aegis/logs/ -name "nightly_*.log" -mtime +30 -delete EOF chmod +x ~/.aegis/scripts/nightly.sh

(crontab -l 2>/dev/null; cat << 'CRON' # AEGIS ZERO ───────────────────────────────── */5 * * * * /bin/bash ~/.aegis/scripts/insurance.sh >> ~/.aegis/logs/insurance.log 2>&1 0 2 * * * /bin/bash ~/.aegis/scripts/nightly.sh >> ~/.aegis/logs/nightly_cron.log 2>&1 0 3 * * 0 /bin/bash ~/.aegis/scripts/backup.sh >> ~/.aegis/logs/backup.log 2>&1 0 * * * * python3 ~/.aegis/mcp/system_server.py health_check >> ~/.aegis/logs/doctor.log 2>&1 # END AEGIS ZERO ────────────────────────────── CRON ) | crontab - crontab -l | grep -c "AEGIS ZERO" && echo "✓ Cron installed"
